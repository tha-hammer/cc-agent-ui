---- MODULE ForkSessionReplay ----
EXTENDS Integers, Sequences, FiniteSets, TLC

CONSTANTS PARENT_ID, NEW_ID, ASSISTANT_UUID

ET_USER   == "user"
ET_ASST   == "assistant"
ET_RESULT == "result"

EventSeq == <<
    [etype |-> ET_USER,   esid |-> NEW_ID, isReplay |-> TRUE,  euuid |-> "r-u1"],
    [etype |-> ET_ASST,   esid |-> NEW_ID, isReplay |-> TRUE,  euuid |-> "r-a1"],
    [etype |-> ET_ASST,   esid |-> NEW_ID, isReplay |-> FALSE, euuid |-> "a2"],
    [etype |-> ET_RESULT, esid |-> NEW_ID, isReplay |-> FALSE, euuid |-> ""]
>>

\* cfg:
\*   CONSTANT PARENT_ID      = "11111111-1111-4111-8111-111111111111"
\*   CONSTANT NEW_ID         = "22222222-2222-4222-8222-222222222222"
\*   CONSTANT ASSISTANT_UUID = "aaaaaaaa-aaaa-4aaa-8aaa-aaaaaaaaaaaa"
\*   INVARIANT StreamTerminates
\*   INVARIANT NewSessionIdObserved
\*   INVARIANT ParentIdNotLeaked
\*   INVARIANT AdapterNeverFails
\*   INVARIANT AdapterOutputWellFormed
\*   INVARIANT ReplayIdentityPreserved

(* --algorithm ForkSessionReplay

variables
    forkValid         \in {TRUE, FALSE},
    eventIndex        = 1,
    curEv             = [etype |-> "", esid |-> "", isReplay |-> FALSE, euuid |-> ""],
    observedSids      = {},
    normalizedMsgs    = <<>>,
    streamDone        = FALSE,
    adapterErr        = FALSE,
    forkErr           = FALSE,
    phase             = "Init";

define

    ValidNm(nm) ==
        /\ nm.provider = "claude"
        /\ nm.nmid     # ""
        /\ nm.nmsid    # ""

    StreamTerminates ==
        (forkValid /\ phase = "Complete") => streamDone

    NewSessionIdObserved ==
        (forkValid /\ phase = "Complete") => (NEW_ID \in observedSids)

    ParentIdNotLeaked ==
        PARENT_ID \notin observedSids

    AdapterNeverFails ==
        forkValid => ~adapterErr

    AdapterOutputWellFormed ==
        \A i \in 1..Len(normalizedMsgs) : ValidNm(normalizedMsgs[i])

    ReplayIdentityPreserved ==
        \A i \in 1..Len(normalizedMsgs) :
            normalizedMsgs[i].kind = "replay" =>
                normalizedMsgs[i].nmid # ""

    AllInvariants ==
        /\ StreamTerminates
        /\ NewSessionIdObserved
        /\ ParentIdNotLeaked
        /\ AdapterNeverFails
        /\ AdapterOutputWellFormed
        /\ ReplayIdentityPreserved

end define;

fair process consumer = "consumer"
begin
    CheckFork:
        if ~forkValid then
            forkErr := TRUE;
            phase   := "ForkRejected";
            goto Terminate;
        end if;
        phase := "Streaming";

    ConsumeLoop:
        while eventIndex <= Len(EventSeq) /\ ~adapterErr do
            curEv        := EventSeq[eventIndex];
            observedSids := observedSids \union {curEv.esid};

            NormalizeStep:
                if curEv.etype = ET_RESULT then
                    streamDone := TRUE;
                    phase      := "Complete";
                    eventIndex := eventIndex + 1;
                elsif curEv.isReplay then
                    if curEv.euuid = "" then
                        adapterErr := TRUE;
                        phase      := "AdapterError";
                    else
                        normalizedMsgs := Append(normalizedMsgs, [
                            nmid     |-> curEv.euuid,
                            nmsid    |-> curEv.esid,
                            provider |-> "claude",
                            kind     |-> "replay",
                            role     |-> "user"
                        ]);
                        eventIndex := eventIndex + 1;
                    end if;
                else
                    normalizedMsgs := Append(normalizedMsgs, [
                        nmid     |-> curEv.euuid,
                        nmsid    |-> curEv.esid,
                        provider |-> "claude",
                        kind     |-> "text",
                        role     |-> "assistant"
                    ]);
                    eventIndex := eventIndex + 1;
                end if;
        end while;

    Terminate:
        skip;
end process;

end algorithm; *)
