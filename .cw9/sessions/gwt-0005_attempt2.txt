CONSTANTS
    RESUME_VALUE,
    CWD_VALUE,
    MODEL_VALUE

(* --algorithm GWT0005ForwardsResume

variables
    resumePresent \in {TRUE, FALSE},
    inputCwd      = CWD_VALUE,
    inputModel    = MODEL_VALUE,
    sdkResume     = "UNSET",
    sdkCwd        = "UNSET",
    sdkModel      = "UNSET",
    phase         = "Init";

define

    ResumeForwarded ==
        (resumePresent /\ phase = "Complete") =>
            sdkResume = RESUME_VALUE

    ResumeAbsent ==
        ((~resumePresent) /\ phase = "Complete") =>
            sdkResume = "UNSET"

    NoClobberCwd ==
        phase = "Complete" =>
            sdkCwd = inputCwd

    NoClobberModel ==
        phase = "Complete" =>
            sdkModel = inputModel

    AllInvariants ==
        ResumeForwarded /\ ResumeAbsent /\ NoClobberCwd /\ NoClobberModel

end define;

fair process mapper = "mapper"
begin
    BuildBase:
        sdkCwd   := inputCwd;
        sdkModel := inputModel;
        phase    := "BaseBuilt";
    CopyResume:
        if resumePresent then
            sdkResume := RESUME_VALUE;
        end if;
    Complete:
        phase := "Complete";
end process;

end algorithm; *)
